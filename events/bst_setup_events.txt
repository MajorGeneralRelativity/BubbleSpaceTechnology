	is_triggered_only = yes
	fire_only_once = yes
	title = "bst_setup.1.name"
	desc = "bst_setup.1.desc"
	immediate = {
		
	}
}
country_event = {
	id = bst_survey.1
	fire_only_once = yes
	hide_window = yes
	
	immediate = {
		set_country_flag = bst_beginning_survey
		begin_event_chain = {
			event_chain = survey_bst_worlds_chain
			target = ROOT
		}
	}
}

ship_event = {
	id = bst_survey.2
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		owner = { has_event_chain = survey_bst_worlds_chain }
		from = {
			OR = {
				is_planet_class = pc_desert
				is_planet_class = pc_tropical
				is_planet_class = pc_arid
				is_planet_class = pc_continental
				is_planet_class = pc_ocean
				is_planet_class = pc_tundra
				is_planet_class = pc_arctic
				is_planet_class = pc_alpine
				is_planet_class = pc_savannah
				is_planet_class = pc_gaia				
			}
		}
	}

	immediate = {
		owner = {
			add_event_chain_counter = {
				event_chain = survey_bst_worlds_chain
				counter = bst_worlds_surveyed
				amount = 1
			}
			country_event = {
				id = bst_survey.3
				days = 1
			}
		}
	}
}

country_event = {
	id = bst_survey.3

	is_triggered_only = yes
	title = "bst_survey.3.name"
	desc = "bst_survey.3.desc"

	trigger = {
		has_completed_event_chain_counter = {
			event_chain = survey_bst_worlds_chain
			counter = bst_worlds_surveyed
		}
	}
	immediate = {
		set_country_flag = bst_completed_beginning_survey
	}
}
